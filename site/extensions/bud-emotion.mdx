## Usage

Add the extension to your config. Make sure it is included after [@roots/bud-babel](/extensions/bud-babel) and [@roots/bud-react](/extensions/bud-react).

<Tabs
  groupId="lang"
  defaultValue="ts"

```js title='bud.config.js' {3,9}
const BudBabelExtension = require('@roots/bud-babel')
const BudReactExtension = require('@roots/bud-react')
const BudEmotionExtension = require('@roots/bud-emotion')

module.exports = app =>
  app.use([
    BudBabelExtension,
    BudReactExtension,
    BudEmotionExtension,
  ])
```

import type {Framework} from '@roots/bud'
import * as BudBabelExtension from '@roots/bud-babel'
import * as BudReactExtension = from '@roots/bud-react'
import * as BudEmotionExtension = from '@roots/bud-emotion'

export default (app: Framework) => 
  app.use([
    BudBabelExtension,
    BudReactExtension,
    BudEmotionExtension,
  ])
